# ===== BACKEND BUILD =====
FROM node:18-alpine AS backend-build

# Set working directory
WORKDIR /app/backend

# Copy package files and install dependencies
COPY backend/package*.json ./
RUN npm ci --only=production

# Copy backend source code
COPY backend/ ./

# ===== FRONTEND BUILD =====
FROM node:18-alpine AS frontend-build

# Set working directory
WORKDIR /app/frontend

# Copy package files and install dependencies
COPY frontend/package*.json ./
RUN npm install --legacy-peer-deps

# Copy frontend source code
COPY frontend/ ./

# Build frontend
RUN npm run build

# ===== CHATBOT BUILD =====
FROM python:3.10-slim AS chatbot-build

# Set working directory
WORKDIR /app/chatbot

# Install dependencies
COPY chatbot/requirements.txt ./
RUN pip install --no-cache-dir -r requirements.txt

# Copy chatbot source code
COPY chatbot/ ./

# ===== FINAL IMAGE =====
FROM node:18-alpine

# Install Python and required packages
RUN apk add --no-cache python3 py3-pip

# Create app directory structure
WORKDIR /app

# Create non-root user for security
RUN addgroup -S appgroup && adduser -S appuser -G appgroup

# Copy backend from build stage
COPY --from=backend-build --chown=appuser:appgroup /app/backend /app/backend

# Copy frontend build from build stage
COPY --from=frontend-build --chown=appuser:appgroup /app/frontend/build /app/frontend/build

# Copy chatbot from build stage
COPY --from=chatbot-build --chown=appuser:appgroup /app/chatbot /app/chatbot

# Create directory for data persistence
RUN mkdir -p /app/data && chown -R appuser:appgroup /app/data

# Set environment variables
ENV NODE_ENV=production
ENV PORT=5000

# Expose ports
EXPOSE 5000 3000 5001

# Switch to non-root user
USER appuser

# Create startup script
COPY --chown=appuser:appgroup <<EOF /app/start.sh
#!/bin/sh
cd /app/backend && node server.js &
cd /app/chatbot && python3 main.py &
wait
EOF

RUN chmod +x /app/start.sh

# Start the application
CMD ["/app/start.sh"]
